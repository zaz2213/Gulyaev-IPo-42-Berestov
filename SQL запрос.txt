
CREATE TABLE app.partners (
  id SERIAL PRIMARY KEY,
  partner_type VARCHAR(5),
  partner_name VARCHAR(50),
  partner_director VARCHAR(120),
  partner_email VARCHAR(120),
  partner_number VARCHAR(12),
  partner_address VARCHAR(240),
  partner_inn VARCHAR,
  partner_rating SMALLINT
);

CREATE TABLE app.product_type (
  id SERIAL PRIMARY KEY,
  product_type VARCHAR(20),
  coaf_product_type REAL
);

CREATE TABLE app.products (
  id SERIAL PRIMARY KEY,
  product_type INTEGER REFERENCES app.product_type(id),
  product_name VARCHAR(120),
  partner_artc INTEGER,
  partner_price REAL
);

CREATE TABLE app.partner_products (
  id SERIAL PRIMARY KEY,
  product INTEGER REFERENCES app.products(id),
  partner INTEGER REFERENCES app.partners(id),
  partner_count INTEGER,
  sale_date DATE
);
